51. 创建 Pod，打标签 run=nginx，使用 nodeSelector 将 Pod 部署到带有 nodeName=nginxnode 的节点

  • 创建用的 yaml 文件

apiVersion: v1
kind: Pod
metadata:
  creationTimestamp: null
  # 达标欧倩
  labels:
    run: nginx
  name: nginx
spec:
  # 选择节点，
  nodeSelector:
    nodeName: nginxnode
  containers:
  - image: nginx
    name: nginx
    resources: {}
  dnsPolicy: ClusterFirst
  restartPolicy: Never
status: {}

1

kubectl create -f pod.yaml
